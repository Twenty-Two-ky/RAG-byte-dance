## 一、重构总览

### 1.1 核心问题

当前项目的本质：

```
当前项目 = LLM + 固定Prompt + ChromaDB检索
```

缺失的Agent能力：

| 能力 | 当前状态 | 重构后 |
|------|---------|--------|
| 自主推理循环 | ❌ 单次调用 | ✅ ReAct Loop |
| 工具调用 | ❌ 无 | ✅ 5个结构化工具 |
| 短期记忆 | ⚠️ Session State | ✅ ConversationMemory |
| 长期记忆 | ❌ 无 | ✅ ChromaDB持久化 |
| 任务规划 | ❌ 无 | ✅ Planner Agent |
| 自我反思 | ❌ 无 | ✅ Reflection机制 |
| 多Agent协同 | ⚠️ 手动触发 | ✅ LangGraph自动编排 |
| 混合检索 | ❌ 仅向量 | ✅ 向量 + BM25 + Re-rank |

from langchain_chroma import Chroma
from langchain_google_genai import GoogleGenerativeAIEmbeddings

from sentence_transformers import CrossEncoder

class RAGEngine:
    """
    升级后的RAG引擎

    变化:
    - TextSplitter → langchain RecursiveCharacterTextSplitter
    - 纯向量检索 → 混合检索(向量+BM25) + Re-ranking
    - 对外接口保持不变(向后兼容)
    """

    def __init__(self, api_key: str, embedding_backend: str = "gemini"):
        # 1. Embedding初始化（支持切换）
        if embedding_backend == "gemini":
            self.embeddings = GoogleGenerativeAIEmbeddings(
                model="models/text-embedding-004",
                google_api_key=api_key
            )
        # 预留: elif embedding_backend == "bge": ...

        # 2. LangChain Chroma封装（替换原来的chromadb原始API）
        self.knowledge_store = Chroma(
            collection_name="company_knowledge",
            embedding_function=self.embeddings,
            persist_directory=DB_PATH
        )
        self.history_store = Chroma(
            collection_name="history_cases",
            embedding_function=self.embeddings,
            persist_directory=DB_PATH
        )

        # 3. 文本切片器（替换自研TextSplitter）
        self.splitter = RecursiveCharacterTextSplitter(
            chunk_size=500,
            chunk_overlap=100,

        self._bm25_docs = []
        self._build_bm25_index()

        # 5. Re-ranking模型（新增）

        self.reranker = CrossEncoder('BAAI/bge-reranker-base')

    def add_knowledge(self, file_obj, summary="", content_text=None, model_name=None):

        新内部流程:
        1. 向量检索 Top-10 (LangChain Chroma)
        2. BM25检索 Top-10
        3. RRF融合 → Top-6

    def get_knowledge_retriever(self, k: int = 3):
        """返回LangChain标准Retriever，供LCEL链使用"""
        return self.knowledge_store.as_retriever(
            search_kwargs={"k": k}
        )

    def hybrid_search(self, query: str, top_k: int = 5) -> list[dict]:
        """新增：混合检索接口，供Agent工具调用"""
        ...

    def parse_file_content(self, file_obj, file_type, model_name=None) -> str:
        """多模态解析 - 接口不变，内部保持Gemini SDK调用（多模态不走LangChain）"""
        ...

    # ---- 私有方法 ----

    def _build_bm25_index(self):
        """构建BM25索引（RAGEngine初始化时调用，添加文档后需rebuild）"""
        all_docs = self.knowledge_store.get()
        if all_docs:
            self._bm25_docs = all_docs['documents']
            self._bm25_index = BM25Okapi([doc.split() for doc in self._bm25_docs])

    def _hybrid_score_fusion(self, vector_results, bm25_results,
                              v_weight=0.7, b_weight=0.3) -> list:
        """RRF融合算法，合并两路检索结果"""
        ...

    def _rerank(self, query: str, candidates: list, top_k: int = 3) -> list:
        """BGE-Reranker重排序"""
        pairs = [[query, doc] for doc in candidates]
        scores = self.reranker.predict(pairs)
        sorted_idx = sorted(range(len(scores)), key=lambda i: scores[i], reverse=True)
        return [candidates[i] for i in sorted_idx[:top_k]]

## 五、LLM Client升级设计

### 5.1 新文件：`core/lc_chain.py`

这是LangChain LCEL链的统一入口，**替代**原有`llm_client.py`中的函数。

```python
# core/lc_chain.py 完整接口定义

from langchain_google_genai import ChatGoogleGenerativeAI
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import StrOutputParser, JsonOutputParser
from langchain_core.runnables import RunnablePassthrough, RunnableParallel


class LCChainManager:
    """
    LCEL链管理器

    所有链都通过 | 管道符组装，这是LangChain的标志性语法。

    面试重点: 能解释 prompt | llm | parser 的执行流程
    """

    def __init__(self, api_key: str, model_name: str):
        self.llm = ChatGoogleGenerativeAI(
            model=model_name.replace("models/", ""),
            google_api_key=api_key,
            temperature=0.7,
            convert_system_message_to_human=True  # Gemini特殊要求
        )
        self.llm_flash = ChatGoogleGenerativeAI(
            model="gemini-1.5-flash",  # 快速模型用于过滤等轻量任务
            google_api_key=api_key,
            temperature=0.3
        )

    # =========== 链1: 测试用例生成链 ===========
    def build_case_generation_chain(self, retriever=None):
        """
        RAG增强的用例生成链

        链结构:
        {rag_context, prd_text} → prompt | llm | str_parser

        如果传入retriever，则自动从知识库检索上下文。
        """
        prompt = ChatPromptTemplate.from_messages([
            ("system", "{system_prompt}"),
            ("human", """
【需求文档】
{prd_text}

【知识库参考规范】
{rag_context}

请生成测试用例（严格JSON格式）:
""")
        ])

        if retriever:
            # 带RAG的链: 自动从知识库检索
            def format_docs(docs):
                return "\n\n<<<RAG_SEP>>>\n\n".join(
                    f"【{doc.metadata.get('source','规范')}】\n{doc.page_content}"
                    for doc in docs
                )

            chain = (
                RunnableParallel({
                    "rag_context": retriever | format_docs,
                    "prd_text": RunnablePassthrough(),
                    "system_prompt": lambda _: PromptManager.CORE_SYSTEM_PROMPT
                })
                | prompt
                | self.llm
                | StrOutputParser()
            )
        else:
            # 不带RAG的链: 手动传入上下文
            chain = prompt | self.llm | StrOutputParser()

        return chain

    # =========== 链2: RAG过滤链（替代原LLM细筛）===========
    def build_rag_filter_chain(self):
        """
        知识提纯链（去噪过滤）

        链结构:
        {query, raw_context} → filter_prompt | llm_flash | str_parser

        注意: 用Flash模型，速度快成本低
        """
        prompt = ChatPromptTemplate.from_template(
            PromptManager.RAG_FILTER_PROMPT
        )
        return prompt | self.llm_flash | StrOutputParser()

    # =========== 链3: 摘要生成链 ===========
    def build_summary_chain(self):
        """
        文档摘要生成链

        链结构:
        content → summary_prompt | llm_flash | str_parser
        """
        prompt = ChatPromptTemplate.from_template(
            PromptManager.SUMMARY_PROMPT + "\n\n内容:\n{content}"
        )
        return prompt | self.llm_flash | StrOutputParser()

    # =========== 链4: 用例精炼链 ===========
    def build_refinement_chain(self):
        """
        用例微调链（多轮对话中使用）

        链结构:
        {user_instruction, rag_context, current_cases} → refine_prompt | llm | str_parser
        """
        prompt = ChatPromptTemplate.from_template(
            PromptManager.get_refinement_prompt("{user_instruction}", "{rag_context}")
        )
        return prompt | self.llm | StrOutputParser()

    # =========== 兼容接口：提供与旧llm_client.py相同的函数签名 ===========

    def chat(self, history: list, user_input: str, system_instruction: str = None):
        """
        向后兼容接口，替代原 get_gemini_chat_response()

        内部用LCEL实现，对外行为不变。
        """
        from langchain_core.messages import HumanMessage, AIMessage, SystemMessage

        messages = []
        if system_instruction:
            messages.append(SystemMessage(content=system_instruction))

        for msg in history:
            role = msg.get('role', '')
            content = str(msg.get('parts', [''])[0]) if msg.get('parts') else ''
            if role == 'user':
                messages.append(HumanMessage(content=content))
            elif role == 'model':
                messages.append(AIMessage(content=content))

        messages.append(HumanMessage(content=str(user_input)))

        response = self.llm.invoke(messages)

        # 返回与原函数相同格式
        updated_history = history + [
            {'role': 'user', 'parts': [str(user_input)]},
            {'role': 'model', 'parts': [response.content]}
        ]
        return response.content, updated_history

    def generate_summary(self, content: str) -> str:
        """向后兼容接口，替代原 generate_summary()"""
        chain = self.build_summary_chain()
        return chain.invoke({"content": content[:8000]})
```

## 六、工具库设计

### 6.1 设计原则

Agent的工具遵循**LangChain BaseTool规范**，这样可以直接被LangChain/LangGraph的Agent调用。

每个工具包含：
- `name`: 工具名（Agent在Prompt中用此名称调用）
- `description`: 工具描述（告诉LLM什么时候该用这个工具）
- `args_schema`: 输入参数的Pydantic模型（类型安全）
- `_run()`: 实际执行逻辑

### 6.2 五个工具定义

```python
# core/agent/tools.py 完整定义

from langchain_core.tools import BaseTool
from pydantic import BaseModel, Field
from typing import Optional

# =========== 工具1: 搜索技术规范 ===========
class SearchSpecsInput(BaseModel):
    query: str = Field(description="搜索关键词，如'登录密码规范'、'支付安全要求'")
    top_k: int = Field(default=3, description="返回条数")

class SearchTechSpecsTool(BaseTool):
    name: str = "search_tech_specs"
    description: str = """
    搜索公司技术规范库。
    当你需要了解某个功能的技术要求、安全规范、性能指标时使用。
    例如: 密码规则、接口要求、兼容性要求等。
    输入: 搜索关键词
    输出: 相关规范文本片段
    """
    args_schema: type = SearchSpecsInput
    rag_engine: object = None  # 注入RAGEngine实例

    def _run(self, query: str, top_k: int = 3) -> str:
        results = self.rag_engine.hybrid_search(query, top_k=top_k)
        if not results:
            return "未找到相关技术规范，请扩大搜索范围。"
        return "\n\n---\n\n".join([r['content'] for r in results])

# =========== 工具2: 搜索历史案例 ===========
class SearchHistoryInput(BaseModel):
    query: str = Field(description="搜索关键词，如'登录功能测试用例'")
    top_k: int = Field(default=2, description="返回条数")

class SearchHistoryCasesTool(BaseTool):
    name: str = "search_history_cases"
    description: str = """
    搜索历史归档的测试用例。
    当你需要参考以前生成过的类似用例时使用。
    可以避免重复劳动，借鉴已验证的测试思路。
    输入: 功能模块关键词
    输出: 历史用例JSON
    """
    args_schema: type = SearchHistoryInput
    rag_engine: object = None

    def _run(self, query: str, top_k: int = 2) -> str:
        # 仅搜索history_collection
        results = self.rag_engine.history_store.similarity_search(query, k=top_k)
        if not results:
            return "暂无相关历史用例。"
        return "\n\n".join([doc.page_content for doc in results])

# =========== 工具3: 边界值计算 ===========
class BoundaryCalcInput(BaseModel):
    field_name: str = Field(description="字段名称，如'密码长度'")
    min_value: int = Field(description="最小值")
    max_value: int = Field(description="最大值")

class CalculateBoundaryTool(BaseTool):
    name: str = "calculate_boundary"
    description: str = """
    计算字段的边界值测试点。
    当PRD中有明确的数值范围限制时使用，自动生成7个边界测试点。
    例如: 密码长度6-20位 → 自动生成5,6,7,13,19,20,21 这7个测试值。
    输入: 字段名、最小值、最大值
    输出: 7个边界测试点及其说明
    """
    args_schema: type = BoundaryCalcInput

    def _run(self, field_name: str, min_value: int, max_value: int) -> str:
        midpoint = (min_value + max_value) // 2
        boundary_points = {
            f"低于最小值(无效)": min_value - 1,
            f"最小值(有效下界)": min_value,
            f"最小值+1(有效)": min_value + 1,
            f"中间值(有效)": midpoint,
            f"最大值-1(有效)": max_value - 1,
            f"最大值(有效上界)": max_value,
            f"超过最大值(无效)": max_value + 1,
        }
        result = f"【{field_name}】边界值测试点（范围: {min_value}-{max_value}）:\n"
        for desc, val in boundary_points.items():
            result += f"  - {desc}: {val}\n"
        return result

from langchain_google_genai import ChatGoogleGenerativeAI
from langchain.agents import AgentExecutor, create_react_agent
from langchain_core.prompts import PromptTemplate

class TestCaseReActAgent:
    """
    真正的ReAct Agent，取代原来的"Prompt → 生成"单步流程。

    核心差异:
    - 旧: LLM被动执行，一次调用返回结果
    - 新: Agent主动规划，多轮工具调用后才输出结果
    """

    def __init__(self, api_key: str, model_name: str,
                 rag_engine, evaluator, memory_system):
        # 初始化LLM
        self.llm = ChatGoogleGenerativeAI(
            model=model_name.replace("models/", ""),

        )

        # 注册工具（注入依赖）

        self.react_prompt = PromptManager.get_react_agent_prompt()

        # 创建ReAct Agent
        self.agent = create_react_agent(
            llm=self.llm,
            tools=self.tools,
            prompt=self.react_prompt
        )

        self.executor = AgentExecutor(
            agent=self.agent,
            tools=self.tools,
            verbose=True,           # 打印推理过程（调试用）

            handle_parsing_errors=True,  # 容错处理

        )

        """
        执行ReAct Agent，返回生成结果。

        返回格式（与原有llm_client接口兼容）:
        {
            "response_text": "AI的分析说明",
            "json_data": [...],  # 测试用例列表

        return self._parse_agent_output(result)

{json.dumps(current_cases, ensure_ascii=False)}

请根据指令修改并输出完整的测试用例JSON数组。
"""
        })

        """解析AgentExecutor的输出，提取JSON和说明文本"""
        output = result.get('output', '')
        json_data = extract_json_from_text(output)

        return {
            "response_text": output,
            "json_data": json_data,
            "thought_chain": result.get('intermediate_steps', [])

# core/agent/graph.py 结构设计

class MultiAgentOrchestrator:
    """
    多Agent编排器。
    对外提供统一接口，内部用LangGraph管理Agent协同。
    """

    def __init__(self, api_key, model_name, rag_engine, evaluator):
        # 初始化各Agent
        self.planner_llm = ...    # gemini-1.5-flash（快速分析）
        self.generator_agent = TestCaseReActAgent(...)  # 主Agent
        self.critic = evaluator   # 复用现有Evaluator
        self.refiner_llm = ...    # gemini-1.5-pro（精细优化）

        # 构建状态图
        self.graph = self._build_graph()

    def _build_graph(self):
        """构建LangGraph状态图"""
        workflow = StateGraph(TestCaseState)

        # 添加节点
        workflow.add_node("planner", self._planner_node)
        workflow.add_node("generator", self._generator_node)
        workflow.add_node("critic", self._critic_node)
        workflow.add_node("refiner", self._refiner_node)

        # 添加边
        workflow.set_entry_point("planner")
        workflow.add_edge("planner", "generator")
        workflow.add_edge("generator", "critic")

        # 条件边（核心：决定是继续优化还是结束）
        workflow.add_conditional_edges(
            "critic",
            self._should_refine,           # 判断函数
            {"refine": "refiner", "end": END}  # 路由映射
        )
        workflow.add_edge("refiner", "generator")

        return workflow.compile()

    def _should_refine(self, state: TestCaseState) -> str:
        """条件判断：是否需要继续优化"""
        score = state['evaluation_report'].get('score', 0)
        iterations = state['iteration_count']
        max_iter = state['max_iterations']

        if score >= 85 or iterations >= max_iter:
            return "end"   # 质量达标或超出迭代上限
        return "refine"    # 继续优化

    def run(self, prd_text: str, rag_context: str = "") -> dict:
        """执行完整的多Agent流程"""
        initial_state = TestCaseState(
            prd_text=prd_text,
            rag_context=rag_context,
            iteration_count=0,
            max_iterations=3,
            is_satisfied=False,
            # 其余字段初始化为None
        )

        final_state = self.graph.invoke(initial_state)

        return {
            "cases": final_state['generated_cases'],
            "score": final_state['evaluation_report']['score'],
            "iterations": final_state['iteration_count']
        }

## 附录B：面试FAQ

**Q: 你为什么用LangGraph而不是直接写循环？**
> LangGraph将工作流可视化为有向图，支持状态共享、条件路由和循环控制。
> 相比手写循环，它的优势是：
> 1. 状态类型安全（TypedDict）
> 2. 条件边让跳转逻辑清晰可读
> 3. 内置检查点（Checkpoint）支持暂停和恢复

**Q: ReAct和普通Prompt有什么本质区别？**
> 普通Prompt是"单轮感知-生成"，AI被动执行；
> ReAct是"多轮规划-行动-观察"循环，AI主动决策：
> 它可以决定"我现在需要搜索规范"，然后真的调用工具，
> 根据工具结果再调整下一步策略，直到任务完成。

**Q: BGE-Reranker和向量检索有什么区别？**
> 向量检索用Bi-Encoder：query和document分别编码，计算点积，速度快但精度有限。
> BGE-Reranker用Cross-Encoder：query和document拼接后联合编码，精度更高但速度慢。
> 因此最优实践是：向量检索召回Top-20（快），Reranker精排取Top-3（准）。
```
